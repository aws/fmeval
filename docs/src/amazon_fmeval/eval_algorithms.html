 84class EvalOutput:
 85    """
 86    The class that contains evaluation scores from `EvalAlgorithmInterface`.
 87
 88    :param eval_name: The name of the evaluation
 89    :param dataset_name: The name of dataset used by eval_algo
 90    :param prompt_template: A template used to compose prompts, only consumed if model_output is not provided in dataset
 91    :param dataset_scores: The aggregated score computed across the whole dataset.
 92    :param category_scores: A list of CategoryScore object that contain the scores for each category in the dataset.
 93    :param output_path: Local path of eval output on dataset. This output contains prompt-response with
 94    record wise eval scores
 95    :param error: A string error message for a failed evaluation.
 96    """
 97
 98    eval_name: str
 99    dataset_name: str
100    dataset_scores: Optional[List[EvalScore]] = None
101    prompt_template: Optional[str] = None
102    category_scores: Optional[List[CategoryScore]] = None
103    output_path: Optional[str] = None
104    error: Optional[str] = None
105
106    def __post_init__(self):  # pragma: no cover
107        """Post initialisation validations for EvalOutput"""
108        assert self.dataset_scores or self.error
109
110        if not self.category_scores:
111            return
112
113        dataset_score_names = [eval_score.name for eval_score in self.dataset_scores]
114        if self.category_scores:
115            for category_score in self.category_scores:
116                assert len(category_score.scores) == len(self.dataset_scores)
117                assert dataset_score_names == [
118                    category_eval_score.name for category_eval_score in category_score.scores
119                ]
120
121    def __eq__(self, other: Type["EvalOutput"]):  # type: ignore[override]
122        try:
123            assert self.eval_name == other.eval_name
124            assert self.dataset_name == other.dataset_name
125            assert self.prompt_template == other.prompt_template
126            assert self.error == other.error
127            assert self.dataset_scores if other.dataset_scores else not self.dataset_scores
128            if self.dataset_scores:  # pragma: no branch
129                assert self.dataset_scores and other.dataset_scores
130                assert len(self.dataset_scores) == len(other.dataset_scores)
131                assert seq(self.dataset_scores).sorted(key=lambda x: x.name).zip(
132                    seq(other.dataset_scores).sorted(key=lambda x: x.name)
133                ).filter(lambda x: x[0] == x[1]).len() == len(self.dataset_scores)
134            assert self.category_scores if other.category_scores else not self.category_scores
135            if self.category_scores:
136                assert seq(self.category_scores).sorted(key=lambda cat_score: cat_score.name).zip(
137                    seq(other.category_scores).sorted(key=lambda cat_score: cat_score.name)
138                ).filter(lambda item: item[0] == item[1]).len() == len(self.category_scores)
139            return True
140        except AssertionError:
141            return False

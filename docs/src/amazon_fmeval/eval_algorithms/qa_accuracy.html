 77def _normalize_text_quac_protocol(text: str) -> str:
 78    """
 79    Inspired by HELM: https://github.com/stanford-crfm/helm/blob/62f817eb695a31e8389e3f7be30609d3f0871837/src/helm/benchmark/metrics/basic_metrics.py
 80    Given a text, normalize it using the SQUAD / QUAC protocol. That is remove punctuations, excess whitespaces and articles, and return the lowercased tokens.
 81    SQUAD (https://worksheets.codalab.org/rest/bundles/0x6b567e1cf2e041ec80d7098f031c5c9e/contents/blob/) and
 82    QuAC benchmarks (https://s3.amazonaws.com/my89public/quac/scorer.py) use this protocol to normalize text before evaluating it.
 83    HELM (https://github.com/stanford-crfm/helm/blob/62f817eb695a31e8389e3f7be30609d3f0871837/src/helm/benchmark/metrics/basic_metrics.py#L116)
 84    and HuggingFace evaluate (https://github.com/huggingface/evaluate/blob/775555d80af30d83dc6e9f42051840d29a34f31b/metrics/squad/compute_score.py#L11)
 85    also use this to normalization procedure.
 86
 87    :param text: The text that needs to be normalized.
 88    :returns: The normalized text.
 89    """
 90
 91    text = text.lower()
 92    text = "".join(character for character in text if character not in ENGLISH_PUNCTUATIONS)
 93    return " ".join([word for word in text.split(" ") if (word != "" and word not in ENGLISH_ARTICLES)])

 96def _f1_score(model_output: str, target_output: str, *, normalize_text: bool = False) -> float:
 97    """
 98    Inspired by the implementation in HELM: https://github.com/stanford-crfm/helm/blob/62f817eb695a31e8389e3f7be30609d3f0871837/src/helm/benchmark/metrics/basic_metrics.py#L182
 99
100    Given the model output and the target output, compute the f1 score between the two.
101    F1-score is the harmonic mean of precision and recall where precision is the number of
102    words in the prediction that are also found in the target output and recall is the number
103    of words in the target output that are also found in the answer. We normalize the text following
104    the QuAC protocol above.
105
106    :param model_output: The output of a model that we want to evaluate.
107    :param target_output: The reference or the "ground truth" output.
108    :param normalize_text: Normalize the text before computing f1.
109    :returns: The F1 score.
110    """
111    if normalize_text:  # pragma: no branch
112        model_output, target_output = (_normalize_text_quac_protocol(text) for text in (model_output, target_output))
113    ret = f_measure(set(model_output.split(" ")), set(target_output.split(" ")))
114    if ret is None:  # pragma: no cover
115        return 0.0
116    else:
117        return float(ret)

120def _exact_match_score(model_output: str, target_output: str) -> float:
121    """
122    Inspired by HELM: https://github.com/stanford-crfm/helm/blob/62f817eb695a31e8389e3f7be30609d3f0871837/src/helm/benchmark/metrics/basic_metrics.py#L137
123    Computes if the two strings exactly match.
124
125    :param model_output: The output of a model that we want to evaluate.
126    :param target_output: The reference or the "ground truth" output.
127    :returns: 0 is the two inputs do not match, else 1.
128    """
129    return float(model_output.strip() == target_output.strip())

132def _quasi_exact_match_score(model_output: str, target_output: str) -> float:
133    """
134    Inspired by HELM: https://github.com/stanford-crfm/helm/blob/62f817eb695a31e8389e3f7be30609d3f0871837/src/helm/benchmark/metrics/basic_metrics.py#L144
135    Computes if the two strings exactly match after normalizing them.
136
137    :param model_output: The output of a model that we want to evaluate.
138    :param target_output: The reference or the "ground truth" output.
139    :returns: 1 if the two strings match after normalization else 0.
140    """
141    return float(
142        _normalize_text_quac_protocol(model_output.strip()) == _normalize_text_quac_protocol(target_output.strip())
143    )

168    def evaluate(
169        self,
170        model: Optional[ModelRunner] = None,
171        dataset_config: Optional[DataConfig] = None,
172        prompt_template: Optional[str] = None,
173        save: bool = False,
174        num_records=100,
175    ) -> List[EvalOutput]:
176        """
177        QA Accuracy evaluate.
178
179        :param model: An instance of ModelRunner which is the model under evaluation
180        :param dataset_config: The config to load the dataset to use for evaluation. If not provided, model will be
181                               evaluated on all built-in datasets configured for this evaluation.
182        :param prompt_template: A template which can be used to generate prompts, optional, if not provided defaults
183            will be used.
184        :param save: If set to true, prompt responses and scores will be saved to file. The output is written to
185                     EvalAlgorithmInterface.EVAL_RESULTS_PATH
186        :param num_records: The number of records to be sampled randomly from the input dataset to perform the
187                            evaluation
188        :returns: List of EvalOutput objects. Current implementation returns only one score.
189        """
190        if dataset_config:
191            dataset_configs = [dataset_config]
192        else:
193            dataset_configs = [DATASET_CONFIGS[dataset_name] for dataset_name in EVAL_DATASETS[self.eval_name]]
194
195        eval_outputs: List[EvalOutput] = []
196        for dataset_config in dataset_configs:
197            dataset = get_dataset(dataset_config, num_records)
198            validate_dataset(dataset, [TARGET_OUTPUT_COLUMN_NAME, MODEL_INPUT_COLUMN_NAME])
199            dataset_prompt_template = None
200            if MODEL_OUTPUT_COLUMN_NAME not in dataset.columns():
201                util.require(model, "No ModelRunner provided. ModelRunner is required for inference on model_inputs")
202                dataset_prompt_template = (
203                    get_default_prompt_template(dataset_config.dataset_name) if not prompt_template else prompt_template
204                )
205                dataset = generate_prompt_column_for_dataset(
206                    prompt_template=dataset_prompt_template,
207                    data=dataset,
208                    model_input_column_name=MODEL_INPUT_COLUMN_NAME,
209                    prompt_column_name=PROMPT_COLUMN_NAME,
210                )
211                assert model  # to satisfy mypy
212                dataset = generate_model_predict_response_for_dataset(
213                    model=model,
214                    data=dataset,
215                    model_input_column_name=PROMPT_COLUMN_NAME,
216                    model_output_column_name=MODEL_OUTPUT_COLUMN_NAME,
217                    model_log_probability_column_name=MODEL_LOG_PROBABILITY_COLUMN_NAME,
218                )
219            with timed_block(f"Computing score and aggregation on dataset {dataset_config.dataset_name}", logger):
220                for eval_score, eval_fn in QA_ACCURACY_SCORES_TO_FUNCS.items():
221
222                    def _generate_eval_scores(df: pd.DataFrame) -> pd.Series:  # pragma: no cover
223                        """
224                        Map function generating the scores for every input record in input dataset
225                        """
226                        return pd.Series(
227                            data=[
228                                self._get_score(
229                                    target_output=row[TARGET_OUTPUT_COLUMN_NAME],
230                                    model_output=row[MODEL_OUTPUT_COLUMN_NAME],
231                                    eval_fn=eval_fn,
232                                )
233                                for index, row in df.iterrows()
234                            ]
235                        )
236
237                    dataset = dataset.add_column(eval_score, _generate_eval_scores)
238                    dataset = dataset.materialize()
239
240                dataset_scores, category_scores = aggregate_evaluation_scores(
241                    dataset, [F1_SCORE, EXACT_MATCH_SCORE, QUASI_EXACT_MATCH_SCORE], agg_method=MEAN
242                )
243
244                eval_outputs.append(
245                    EvalOutput(
246                        eval_name=self.eval_name,
247                        dataset_name=dataset_config.dataset_name,
248                        prompt_template=dataset_prompt_template,
249                        dataset_scores=dataset_scores,
250                        category_scores=category_scores,
251                        output_path=generate_output_dataset_path(
252                            path_to_parent_dir=self._eval_results_path,
253                            eval_name=self.eval_name,
254                            dataset_name=dataset_config.dataset_name,
255                        ),
256                    )
257                )
258            if save:
259                save_dataset(
260                    dataset=dataset,
261                    score_names=list(QA_ACCURACY_SCORES_TO_FUNCS.keys()),
262                    path=generate_output_dataset_path(
263                        path_to_parent_dir=self._eval_results_path,
264                        eval_name=self.eval_name,
265                        dataset_name=dataset_config.dataset_name,
266                    ),
267                )
268
269        return eval_outputs

271    def _get_score(
272        self, target_output: str, model_output: str, eval_fn: Callable[..., float], **fn_kwargs: Any
273    ) -> float:
274        """
275        Method to generate accuracy score for a target_output and model_output

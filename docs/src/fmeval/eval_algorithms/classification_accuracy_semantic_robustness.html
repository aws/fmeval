289    def evaluate_sample(
290        self,
291        model_input: str,
292        model: ModelRunner,
293        target_output: str,
294        model_output: Optional[str] = None,
295        prompt_template: str = DEFAULT_PROMPT_TEMPLATE,
296    ) -> List[EvalScore]:  # type: ignore[override]
297        """
298        Evaluate a single record for Classification Accuracy Semantic Robustness.
299
300        :param model_input: text input for model
301        :param model: An instance of ModelRunner which is the model under evaluation
302        :param target_output: The expected responses from the model
303        :param model_output: The output of a model that we want to evaluate
304        :param prompt_template: A template which can be used to compose prompt using model_input
305        :returns: A List of EvalScores computed for prompts and responses.
306        """
307        util.require(
308            model_input,
309            "Missing required input: model_input, for ClassificationAccuracySemanticRobustness evaluate_sample",
310        )
311        util.require(
312            model,
313            "Missing required input: model i.e. ModelRunner, for ClassificationAccuracySemanticRobustness evaluate_sample",
314        )
315        util.require(
316            target_output,
317            "Missing required input: target_output, for " "ClassificationAccuracySemanticRobustness evaluate_sample",
318        )
319
320        prompt_composer = PromptComposer(prompt_template)
321        original_prompt = prompt_composer.compose(model_input)
322        original_model_output = model_output if model_output else model.predict(original_prompt)[0]
323
324        perturbation = PERTURBATION_TYPE_TO_HELPER_CLASS[self._eval_algorithm_config.perturbation_type]()
325
326        perturbed_inputs = perturbation.perturb(
327            text=model_input,
328            config=self._perturbation_config,
329            num_perturbations=self._eval_algorithm_config.num_perturbations,
330        )
331
332        perturbed_input_prompts = [prompt_composer.compose(perturbed_input) for perturbed_input in perturbed_inputs]
333        perturbed_input_outputs = [model.predict(prompt)[0] for prompt in perturbed_input_prompts]
334
335        original_classification_accuracy_scores = self._classification_accuracy_eval_algo.evaluate_sample(
336            target_output=target_output, model_output=original_model_output
337        )
338
339        perturbed_outputs_classification_accuracy_scores = defaultdict(lambda: [])
340        for perturbed_input_output in perturbed_input_outputs:
341            accuracy_scores = self._classification_accuracy_eval_algo.evaluate_sample(
342                target_output=target_output, model_output=perturbed_input_output
343            )
344            for accuracy_score in accuracy_scores:
345                perturbed_outputs_classification_accuracy_scores[accuracy_score.name].append(accuracy_score)
346
347        delta_scores = [
348            EvalScore(
349                name=PREFIX_FOR_DELTA_SCORES + original_score.name,
350                value=generate_mean_delta_score(
351                    original_score, perturbed_outputs_classification_accuracy_scores[original_score.name]
352                ),
353            )
354            for original_score in original_classification_accuracy_scores
355        ]
356
357        return original_classification_accuracy_scores + delta_scores

63def convert_model_output_to_label(model_output: str, valid_labels: List[str]) -> str:
64    """Convert model output to string class label. The model is expected to return a label directly (if it has a
65    classification head), or a string containing a label (if it has a language modelling head). In the latter case we
66    strip any additional text (e.g. "The answer is 2." --> "2"). If no valid labels is contained in the
67    `model_output` an "unknown" label is returned. Users can define other `converter_fn`s, e.g. to translate a text
68    label to string ("NEGATIVE" --> "0").
69
70    :param model_output: Value returned by the model.
71    :param valid_labels: Valid labels.
72    :return: `model_output` transformed into a label
73    """
74    # normalise to lowercase & strip
75    valid_labels = [label.lower().strip() for label in valid_labels]
76
77    response_words = model_output.split(" ")
78    predicted_labels = [word.lower().strip() for word in response_words if word.lower().strip() in valid_labels]
79    # if there is more than one label in the model output we pick the first
80    string_label = predicted_labels[0] if predicted_labels else UNKNOWN_LABEL
81
82    return string_label

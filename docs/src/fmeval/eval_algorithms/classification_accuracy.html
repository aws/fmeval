 65def convert_model_output_to_label(model_output: str, valid_labels: List[str]) -> str:
 66    """Convert model output to string class label. The model is expected to return a label directly (if it has a
 67    classification head), or a string containing a label (if it has a language modelling head). In the latter case we
 68    strip any additional text (e.g. "The answer is 2." --> "2"). If no valid labels is contained in the
 69    `model_output` an "unknown" label is returned. Users can define other `converter_fn`s, e.g. to translate a text
 70    label to string ("NEGATIVE" --> "0").
 71
 72    :param model_output: Value returned by the model.
 73    :param valid_labels: Valid labels.
 74    :return: `model_output` transformed into a label
 75    """
 76    # normalise to lowercase & strip
 77    valid_labels = [label.lower().strip() for label in valid_labels]
 78
 79    response_words = model_output.split(" ")
 80    predicted_labels = [word.lower().strip() for word in response_words if word.lower().strip() in valid_labels]
 81    # if there is more than one label in the model output we pick the first
 82    string_label = predicted_labels[0] if predicted_labels else UNKNOWN_LABEL
 83
 84    return string_label

274    def _get_score(self, y_true, y_pred, eval_fn: Callable[..., float]) -> float:
275        """
276        Method to generate accuracy score
277        :param y_true: Ground truth (correct) target values.
278        :param y_pred: Estimated targets as returned by a classifier.
279        :param eval_fn: Score evaluate function.
280        :returns: Computed score
281        """
282        if eval_fn == recall_score or eval_fn == precision_score:
283            return eval_fn(y_true, y_pred, average=self._eval_algorithm_config.multiclass_average_strategy)
284        return eval_fn(y_true, y_pred)

 31class BaseHelperModel(ABC):
 32    """
 33    Base class for 3P helper model invoker. Note: These Helper models are inherently
 34    Machine learning models being used by Evaluation algorithms.
 35    """
 36
 37    @abstractmethod
 38    def get_helper_scores(self, text_input: str) -> Any:
 39        """
 40        Method to invoke helper model
 41        :param text_input: model text input
 42        :returns: model output
 43        """

 46class ToxigenHelperModel(BaseHelperModel):
 47    """
 48    Helper model for toxigen model: https://huggingface.co/tomh/toxigen_roberta/tree/main
 49    """
 50
 51    TOXIGEN_MODEL_NAME = "tomh/toxigen_roberta"
 52    COLUMN_NAME = MODEL_OUTPUT_COLUMN_NAME
 53
 54    def __init__(self, column_name: str = COLUMN_NAME):
 55        """
 56        Constructor to locally load the helper model for inference.
 57
 58        :param column_name: column name used to fetch input texts in __call__ method
 59        """
 60        self._model = pipeline("text-classification", model=self.TOXIGEN_MODEL_NAME)
 61        self._column_name = column_name
 62
 63    def get_helper_scores(self, text_input: List[str]) -> Dict[str, List[float]]:  # type: ignore[override]
 64        """
 65        Method to get scores from ToxigenHelper
 66        :param text_input: list of text inputs for the model
 67        :returns: dict with key as score name and value being list of scores for text inputs
 68
 69        Note: Toxigen scores are for label: LABEL_1
 70        """
 71        inference_output = self._model(text_input)
 72        result = {
 73            TOXIGEN_SCORE_NAME: [x["score"] if x["label"] == "LABEL_1" else 1.0 - x["score"] for x in inference_output]
 74        }
 75        return result
 76
 77    def __call__(self, batch: Dict[str, np.ndarray]) -> Dict[str, np.ndarray]:
 78        """
 79        Call method to allow using this helper as a ray actor.
 80
 81        :param batch: batch of data to be scored.
 82        :return: batch with scores added to it.
 83        """
 84        scores = self.get_helper_scores(batch[self._column_name].tolist())
 85
 86        for key, value in scores.items():
 87            batch.update({key: np.array(value)})
 88        return batch
 89
 90    @staticmethod
 91    def get_score_names() -> List[str]:
 92        """
 93        Util method to return name of scores generated by helper model
 94        :returns: List of score names
 95        """
 96        return [TOXIGEN_SCORE_NAME]

150@ray.remote(num_cpus=1)
151class BertscoreHelperModel(BaseHelperModel):
152    """
153    BERTscore is a similarity-based metric that compares the embedding of the prediction and target sentences
154    under a (learned) model, typically, from the BERT family.
155    This score may lead to increased flexibility compared to rouge and METEOR in terms of rephrasing since
156    semantically similar sentences are (typically) embedded similarly.
157
158    https://huggingface.co/spaces/evaluate-metric/bertscore
159
160    Note: we specify that this Ray actor requires num_cpus=1 in order to limit the number of concurrently
161    running tasks or actors to avoid out of memory issues.
162    See https://docs.ray.io/en/latest/ray-core/patterns/limit-running-tasks.html#core-patterns-limit-running-tasks
163    for a detailed explanation.
164    """
165
166    def __init__(self, model_type: str):  # pragma: no cover
167        """
168        Default constructor
169
170        :param model_type: Model type to be used for bertscore
171        """
172        self._bertscore = hf_evaluate.load("bertscore")
173        self._model_type = model_type
174
175        # Dummy call to download the model within constructor
176        self._bertscore.compute(
177            predictions=["dummy_prediction"],
178            references=["dummy_reference"],
179            model_type=self._model_type,
180        )
181
182    def get_helper_scores(self, target_output: str, model_output: str) -> float:  # type: ignore[override]
183        """
184        Method to invoke the concerned model and get bertscore
185        :param target_output: Reference text
186        :model_output: Model prediction text
187        """
188        # Note: the following code is covered by unit tests,
189        # but since it gets executed by Ray, Mypy marks it
190        # as not covered.
191        return self._bertscore.compute(  # pragma: no cover
192            predictions=[model_output],
193            references=[target_output],
194            model_type=self._model_type,
195        )["f1"][0]
